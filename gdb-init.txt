# 1. Connect to the OpenOCD server.
target remote localhost:3333

# 2. Reset the Pi and halt it at the very beginning.
monitor reset init

# 3. Load your program's symbols and code into the Pi's RAM.
load

# 4. Set a breakpoint at the program's entry point.
break _start

# 5. Start the program. It will run and immediately stop at your breakpoint.
continue
